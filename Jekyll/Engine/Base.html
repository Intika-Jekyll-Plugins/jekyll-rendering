def render_error(err, renderer = nil, trace = Rendering::ERROR_TRACE)
  msg = [self.class.name.split('::').last]
  msg << "[#{renderer}]" if renderer

  msg << 'Exception:' << err
  msg << "(in #{data['layout'] || '(top)'})"

  msg << "(from #{err.backtrace[0, trace].join(', ')})" if trace > 0

  msg.join(' ')
end
